CREATE TABLE IF NOT EXISTS UDPONLY (
   PK serial primary key,
   TimeStamp Timestamp,
   SMac varchar(17),
   DMac varchar(17),
   SIp varchar(15),
   DIp varchar(15),
   SPort int,
   DPort int,
   TTL int,
   TOS int,
   ID int,
   IHL int,
   DLen int,
   Proto int
);

CREATE TABLE IF NOT EXISTS TCPONLY (
   PK serial primary key,
   TimeStamp Timestamp,
   SMac varchar(17),
   DMac varchar(17),
   SIp varchar(15),
   DIp varchar(15),
   SPort int,
   DPort int,
   TTL int,
   TOS int,
   ID int,
   IHL int,
   DLen int,
   Proto int
);

CREATE TABLE IF NOT EXISTS ARPONLY (
   PK serial primary key,
   TimeStamp Timestamp,
   SMac varchar(17),
   DMac varchar(17),
   HWsrc varchar(17),
   HWdst varchar(17),
   PSrc varchar(15),
   PDst varchar(15)
);

CREATE TABLE IF NOT EXISTS IGMPONLY (
   PK serial primary key,
   TimeStamp Timestamp,
   SMac varchar(17),
   DMac varchar(17),
   SIp varchar(15),
   DIp varchar(15),
   TTL int,
   TOS int,
   ID int,
   IHL int,
   DLen int,
   Proto int
);

CREATE TABLE IF NOT EXISTS POWER (
   PK serial primary key,
   IP varchar(15),
   TimeStamp Timestamp,
   Power Numeric(7,4)
);

CREATE TABLE IF NOT EXISTS HOSTDATA (
   PK serial primary key,
   TimeStamp Timestamp,
   totalRam int,
   usedRam int,
   cpuPercent decimal,
   cpuTemp decimal,
   cpuVolts decimal,
   totalTasks int,
   runningTasks int,
   sleepingTasks int,
   stoppedTasks int, 
   zombieTasks int
);
